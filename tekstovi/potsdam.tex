	\noindent \p{日本 の }\p{降伏 の }\p{ため の }\p{定義 }\p{および }\p{規約}\\[-2em]
	\begin{enumerate}	
		\item \p{我々 }\p{合衆 国 }\p{大統領 、 }\p{中華 }\p{民国 }\p{政府 }\p{主席 、 }\p{及び }\p{英国 }\p{総理 }\p{大臣 は 、 }\p{我々 の }\p{数 }\p{億 の }\p{国民 を }\p{代表 }\p{し }\p{協議 の }\p{上 、 }\p{日本 国 に }\p{対し }\p{戦争 を }\p{終結 }\p{する }\p{機会 を }\p{与える }\p{こと で }\p{一致 }\p{し た 。}
		
		\item \p{三 ヶ国 の }\p{軍隊 は }\p{増強 を }\p{受け 、 }\p{日本 に }\p{最後 の }\p{打撃 を }\p{加える }\p{用意 を }\p{既に }\p{整え た 。 }\p{この }\p{軍事 力 は 、 }\p{日本 国 の }\p{抵抗 が }\p{止まる まで 、 }\p{同国 に }\p{対する }\p{戦争 を }\p{遂行 }\p{する }\p{一切 の }\p{連合 国 の }\p{決意 に }\p{より }\p{支持 }\p{さ れ }\p{且つ }\p{鼓舞 }\p{さ れる 。}
		
		\item \p{世界 の }\p{自由 な }\p{人民 に }\p{支持 }\p{さ れ た }\p{この }\p{軍事 力 }\p{行使 は 、 }\p{ナチス }\p{・}\p{ドイツ に }\p{対し て }\p{適用 }\p{さ れ た }\p{場合 に }\p{ドイツ と }\p{ドイツ }\p{軍 に }\p{完全 に }\p{破壊 を }\p{もたらし た }\p{こと が }\p{示す }\p{よう に 、 }\p{日本 と }\p{日本 }\p{軍 が }\p{完全 に }\p{壊滅 }\p{する }\p{こと を }\p{意味 }\p{する 。}
		
		\item \p{日本 が 、 }\p{無 }\p{分別 な }\p{打算 に }\p{より }\p{自国 を }\p{滅亡 の }\p{淵 に }\p{追い詰め た }\p{軍国 }\p{主義 者 の }\p{指導 を }\p{引き続き }\p{受ける か 、 }\p{それ と も }\p{理性 の }\p{道 を }\p{歩む か を }\p{選ぶ べき }\p{時 が }\p{到来 }\p{し た の だ 。}
		
		\item \p{我々 の }\p{条件 は }\p{以下 の }\p{条文 で }\p{示す }\p{とおり で }\p{あり 、 }\p{これ に }\p{つい て は }\p{譲歩 }\p{せ ず 、 }\p{我々 が }\p{ここ から }\p{外れる }\p{こと も }\p{又 }\p{ない 。 }\p{執行 の }\p{遅れ は }\p{認め ない 。}
		
		\item \p{日本 }\p{国民 を }\p{欺い て }\p{世界 }\p{征服 に }\p{乗り出す }\p{過ち を }\p{犯さ せ た }\p{勢力 を }\p{永久 に }\p{除去 }\p{する 。 }\p{無 }\p{責任 な }\p{軍国 }\p{主義 が }\p{世界 から }\p{駆逐 }\p{さ れる まで は 、 }\p{平和 と }\p{安全 と }\p{正義 の }\p{新 }\p{秩序 も }\p{現れ }\p{得 ない から で }\p{ある 。}
		
		\item \p{第 }\p{六 条 の }\p{新 }\p{秩序 が }\p{確立 }\p{さ れ 、 }\p{戦争 }\p{能力 が }\p{失わ れ た }\p{こと が }\p{確認 }\p{さ れる }\p{時 まで は 、 }\p{我々 の }\p{指示 }\p{する }\p{基本 的 }\p{目的 の }\p{達成 を }\p{確保 }\p{する }\p{ため 、 }\p{日本 国 }\p{領域 内 の }\p{諸 }\p{地点 は }\p{占領 }\p{さ れる べき }\p{もの と }\p{する 。}
		
		\item \p{カイロ }\p{宣言 の }\p{条項 は }\p{履行 }\p{さ れる べき で }\p{あり 、 }\p{又 }\p{日本 国 の }\p{主権 は }\p{本州 、 }\p{北海道 、 }\p{九州 }\p{及び }\p{四国 }\p{ならび に }\p{我々 の }\p{決定 }\p{する }\p{諸 }\p{小島 に }\p{限ら れ なけれ ば }\p{なら ない 。}
		
		\item \p{日本 }\p{軍 は }\p{武装 }\p{解除 }\p{さ れ た }\p{後 、 }\p{各自 の }\p{家庭 に }\p{帰り }\p{平和 }\p{・}\p{生産 的 に }\p{生活 }\p{出来る }\p{機会 を }\p{与え られる 。}
		
		\item \p{我々 の }\p{意志 は }\p{日本 人 を }\p{民族 と }\p{し て }\p{奴隷 化 }\p{し 、 }\p{また }\p{日本 }\p{国民 を }\p{滅亡 }\p{さ せよう と }\p{する }\p{もの で は }\p{ない が 、 }\p{日本 に }\p{おけ る }\p{捕虜 }\p{虐待 を }\p{含む }\p{一切 の }\p{戦争 }\p{犯罪 人 は }\p{処罰 }\p{さ れる べき で }\p{ある 。 }\p{日本 }\p{政府 は }\p{日本 国 }\p{国民 に }\p{おけ る }\p{民主 }\p{主義 的 }\p{傾向 の }\p{復活 を }\p{強化 }\p{し 、 }\p{これ を }\p{妨げる }\p{あらゆる }\p{障碍 は }\p{排除 }\p{する べき で }\p{あり 、 }\p{言論 、 }\p{宗教 }\p{及び }\p{思想 の }\p{自由 }\p{並び に }\p{基本 的 }\p{人権 の }\p{尊重 は }\p{確立 }\p{さ れる べき で }\p{ある 。}
		
		\item \p{日本 は }\p{経済 }\p{復興 }\p{し 、 }\p{課さ れ た }\p{賠償 の }\p{義務 を }\p{履行 }\p{する }\p{ため の }\p{生産 }\p{手段 、 }\p{戦争 と }\p{再 }\p{軍備 に }\p{関わら ない }\p{もの が }\p{保有 }\p{出来る 。 }\p{また }\p{将来 的 に は }\p{国際 }\p{貿易 に }\p{復帰 が }\p{許可 }\p{さ れる 。}
		
		\item \p{日本 国 }\p{国民 が }\p{自由 に }\p{表明 }\p{し た }\p{意志 に }\p{よる }\p{平和 的 }\p{傾向 の }\p{責任 }\p{ある }\p{政府 の }\p{樹立 を }\p{求める 。 }\p{この }\p{項目 }\p{並び に }\p{すでに }\p{記載 }\p{し た }\p{条件 が }\p{達成 }\p{さ れ た }\p{場合 に }\p{占領 }\p{軍 は }\p{撤退 }\p{する べき で }\p{ある 。}
		
		\item \p{我々 は }\p{日本 }\p{政府 が }\p{全 }\p{日本 }\p{軍 の }\p{即時 }\p{無 }\p{条件 }\p{降伏 を }\p{宣言 }\p{し 、 }\p{また }\p{その }\p{行動 に }\p{つい て }\p{日本 }\p{政府 が }\p{十分 に }\p{保障 }\p{する }\p{こと を }\p{求める 。 }\p{これ }\p{以外 の }\p{選択 肢 は }\p{迅速 }\p{且つ }\p{完全 なる }\p{壊滅 が }\p{ある のみ で }\p{ある 。}
	\end{enumerate}
